    public CodeGenerator(String source, String packageName, String className,
                         Map<String, String> symbolTable, Map<String, Integer> localIndex,
                         int numLocals, Map<AlgolParser.ExprContext, String> exprTypes,
                         Map<String, int[]> arrayBounds,
                         Map<String, SymbolTableBuilder.ProcInfo> procedures) {

        // Initialize infrastructure
        this.context = new ContextManager(source, packageName, className, procedures, exprTypes);
        this.context.setSymbolTable(symbolTable);
        this.context.setLocalIndex(localIndex);
        this.context.setNumLocals(numLocals);
        this.context.setArrayBounds(arrayBounds);

        // Initialize delegates
        this.exprGen = new ExpressionGenerator();
        this.exprGen.setContext(context);

        this.stmtGen = new StatementGenerator(this.exprGen);
        this.stmtGen.setContext(context);

        this.procGen = new ProcedureGenerator(this.exprGen, this.stmtGen);
        this.procGen.setContext(context);

        this.mainCode = new StringBuilder();
        this.activeOutput = mainCode;
    }
